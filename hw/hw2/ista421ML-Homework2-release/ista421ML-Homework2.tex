\begin{verbatim}
>>> m = numpy.matrix([1,2,3]) # this creates a matrix with one row
matrix([[1, 2, 3]]) 
>>> m.transpose()   # transposition makes returns a column vector
matrix([[1],
        [2],
        [3]])

matrix([[1],
        [2],
